package com.prismamp.todopago.controllers;

import java.util.Optional;

import javax.validation.Valid;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.cache.annotation.CacheConfig;
import org.springframework.cache.annotation.CacheEvict;
import org.springframework.cache.annotation.Cacheable;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.bind.annotation.RequestParam;

import com.google.common.collect.Iterables;
import com.prismamp.todopago.exceptions.BusinessException;
import com.prismamp.todopago.exceptions.ResourceNotFoundException;
import com.prismamp.todopago.model.abstracts.GenericParameter;
import com.prismamp.todopago.repository.GenericRepository;
import com.prismamp.todopago.utils.ParametersUtils;

@CacheConfig(cacheNames="entities", keyGenerator="myKeyGenerator") 
public class AbstractController<E extends GenericParameter> {

	@Autowired
	GenericRepository<E> repository;

	@Cacheable(cacheManager = "springCM")
	@GetMapping("/{id}")
	public E findById(@PathVariable("id") Integer id) {
		return Optional.ofNullable(repository.findOne(id))
				.orElseThrow(() -> new ResourceNotFoundException(" No se encuentra con id:" + id));

	}

	@Cacheable
	@GetMapping("/")
	public Iterable<E> findAll() {
		return repository.findAll();

	}
	@Cacheable
	@GetMapping("/active")
	public Iterable<E> findActive() {
		return repository.findAllActivate();

	}

	@Cacheable
	@GetMapping("/inactive")
	public Iterable<E> findInActive() {
		return repository.findAllInactivate();

	}

	@GetMapping
	@Cacheable
	public Iterable<E> getByCode(@RequestParam("code") Optional<String> code) {
		Iterable<E> result;
		if (code.isPresent()) {
			result = repository.findByCode(code.get());
		} else {
			result = repository.findAll();
		}
		return result;
	}

	@PostMapping
	@CacheEvict(allEntries = true)
	public E post(@RequestBody @Valid E e) {
		if (!Iterables.isEmpty(repository.findByCode(e.getCode()))) {
			throw new BusinessException("El codigo" + e.getCode() + " ya se encuentra en uso");
		}
		return repository.save(e);
	}

	@PutMapping("/{id}")
	@CacheEvict(allEntries = true)
	public E put(@PathVariable("id") Integer id, @RequestBody @Valid E inputMode) {
		E inputOriginal = Optional.of(repository.findOne(id)).orElseThrow(
				() -> new ResourceNotFoundException("No existe ese id .. cambiar por bussisnes exception"));
		ParametersUtils.copy(inputMode, inputOriginal);
		return repository.save(inputOriginal);
	}

	@RequestMapping(value = "/{id}", method = RequestMethod.DELETE)
	@CacheEvict(allEntries = true)
	public void delete(@PathVariable Integer id) {
		repository.delete(id);
	}
}
